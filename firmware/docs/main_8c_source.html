644 /**************************************************************************//**
645  * @brief
646  * Function to handle encoder rotation logic.
647  *****************************************************************************/
648 void checkEncoder(void)
649 {
650  int16_t steps = 0;
651  uint16_t encValue = LL_TIM_GetCounter(TIM3);
652 
653  if (encValue != settings.oldTIM3)
654  {
655  /* Make sure we get a stable value */
656  while ((encValue % 4) > 0) encValue++;
657 
658  /* Calculate steps */
659  steps = encValue - settings.oldTIM3;
660  steps /= 4;
661 
662  /* Logic for circular pages */
663  if (steps > 0)
664  {
665  for (uint8_t counter = 0; counter < steps; counter++)
666  {
667  /* Rotate LEFT display pages (if selected) */
668  if (settings.selectedDisplay == 0)
669  {
670  if (settings.leftPage < 2) settings.leftPage++;
671  else settings.leftPage = 0;
672 
673  /* Indicate that a certain page needs to be updated */
674  if (settings.leftPage == 0)
675  {
676  cData.newPow = 1;
677  cData.newMaxVolt = 1;
678  cData.newMinVolt = 1;
679  }
680  else if (settings.leftPage == 1)
681  {
682  cData.newPow = 1;
683  cData.newMaxCurr = 1;
684  cData.newMinCurr = 1;
685  }
686  else if (settings.leftPage == 2)
687  {
688  cData.newPow = 1;
689  cData.newAh = 1;
690  cData.newWh = 1;
691  }
692  }
693 
694  /* Rotate RIGHT display pages (if selected) */
695  if (settings.selectedDisplay == 1)
696  {
697  /* Change period (if selected) */
698  if (settings.changePeriod == 1)
699  {
700  if (settings.measPeriodIndex < (LUT_period_s_length - 1))
701  {
702  settings.measPeriodIndex++;
703  settings.remMeasPeriodS = LUT_period_s[settings.measPeriodIndex];
704  settings.measPeriodS = LUT_period_s[settings.measPeriodIndex];
705  settings.periodChanged = 1;
