109 /**************************************************************************//**
110  * @brief
111  * This function monitors USART1/2/3 buffer filling indication and
112  * moves data around accordingly.
113  *****************************************************************************/
114 void USART_HandleContinuousReception(void)
115 {
116  /* Checks if USART1 Buffer full indication has been set */
117  if (uwBufferReadyIndication_USART1 != 0)
118  {
119  /* Call user Callback in charge of consuming data from filled USART2 buffer */
120  SEGGER_RTT_printf(0, "Current USART1 RX buffer full: [%s], Reception will go on in alternate buffer.\n\r", pBufferReadyForUser_USART1);
121 
122  /* Signal logic that the logger has correctly started up */
123  settings.loggerReady = 1;
124 
125  /* Reset USART1 indication */
126  uwBufferReadyIndication_USART1 = 0;
127  }
128 
129  /* Checks if USART2 Buffer full indication has been set */
130  if (uwBufferReadyIndication_USART2 != 0)
131  {
132  data.voltage = bytes_to_float(pBufferReadyForUser_USART2[6], pBufferReadyForUser_USART2[5], pBufferReadyForUser_USART2[4], pBufferReadyForUser_USART2[3]);
133 
134  if (data.voltage < 99.999) /* Meter reads max 50V ... */
135  {
136  if (data.oldVolt != data.voltage)
137  {
138  data.oldVolt = data.voltage;
139  data.newVolt = 1;
140  }
141 
142  /* Update min/max fields if necessary */
143  if (data.voltage < data.minVoltage)
144  {
145  data.minVoltage = data.voltage;
146  data.newMinVolt = 1;
147  }
148  else if (data.firstVolt)
149  {
150  data.minVoltage = data.voltage;
151  data.newMinVolt = 1;
152  data.firstVolt = 0;
153  }
154 
155  if (data.voltage > data.maxVoltage)
156  {
157  data.maxVoltage = data.voltage;
158  data.newMaxVolt = 1;
159  }
160 
161  data.voltReceived = 1;
162  }
163 
164  /* Reset USART2 indication */
165  uwBufferReadyIndication_USART2 = 0;
166 
167  }
168 
169  /* Checks if USART3 Buffer full indication has been set */
170  if (uwBufferReadyIndication_USART3 != 0)
171  {
172  data.current = bytes_to_float(pBufferReadyForUser_USART3[6], pBufferReadyForUser_USART3[5], pBufferReadyForUser_USART3[4], pBufferReadyForUser_USART3[3]);
173 
174  if (data.current < 9.9999) /* Meter reads max 5A ... */
175  {
176  if (data.oldCurr != data.current)
177  {
178  data.oldCurr = data.current;
179  data.newCurr = 1;
180  }
181 
182  /* Update min/max fields if necessary */
183  if (data.current < data.minCurrent)
184  {
185  data.minCurrent = data.current;
186  data.newMinCurr = 1;
187  }
188  else if (data.firstCurr)
189  {
190  data.minCurrent = data.current;
191  data.newMinCurr = 1;
192  data.firstCurr = 0;
193  }
194 
195  if (data.current > data.maxCurrent)
196  {
197  data.maxCurrent = data.current;
198  data.newMaxCurr = 1;
199  }
200 
201  data.currReceived = 1;
202  }
203 
204  /* Reset USART13indication */
205  uwBufferReadyIndication_USART3 = 0;
206  }
207 
208  /* Set values if the voltmeter is powered down */
209  if (!LL_GPIO_IsInputPinSet(V_ENin_GPIO_Port, V_ENin_Pin))
210  {
211  data.voltage = 0.0;
212 
213  if (data.oldVolt != data.voltage)
214  {
215  data.oldVolt = data.voltage;
216  data.newVolt = 1;
217  }
218 
219  data.voltReceived = 1;
220  }
221 
222  /* Set values if the current meter is powered down */
223  if (!LL_GPIO_IsInputPinSet(A_ENin_GPIO_Port, A_ENin_Pin))
224  {
225  data.current = 0.0;
226 
227  if (data.oldCurr != data.current)
228  {
229  data.oldCurr = data.current;
230  data.newCurr = 1;
231  }
232 
233  data.currReceived = 1;
234  }
235 }

238 /**************************************************************************//**
239  * @brief
240  * Send a byte (`uint8_t`) command to USART2.
241  *
242  * @details
243  * This method also resets `uwNbReceivedChars_USART2` just in case.
244  *
245  * @param[in] command
246  * Pointer to the byte-array containing the commands.
247  *
248  * @param[in] bytes
249  * Amount of bytes to send.
250  *****************************************************************************/
251 void USART2_SendCommand(uint8_t *command, uint8_t bytes)
252 {
253  /* Send characters one per one, until last char to be sent */
254  for (uint32_t index = 0; index < bytes; index++)
255  {
256  /* Wait for TXE flag to be raised */
257  while (!LL_USART_IsActiveFlag_TXE(USART2));
258 
259  /* Write character in Transmit Data register.
260  TXE flag is cleared by writing data in DR register */
261  LL_USART_TransmitData8(USART2, command[index]);
262  }
263 
264  /* Wait for TC flag to be raised for last char */
265  while (!LL_USART_IsActiveFlag_TC(USART2));
266 
267  /* Reset buffer mechanism */
268  uwNbReceivedChars_USART2 = 0;
269 // uwBufferReadyIndication_USART2 = 0; // TODO Maybe use this when using contiuous output?
270 }

497 /**************************************************************************//**
498  * @brief
499  * Print a string (char array) to USARTx.
500  *
501  * @note
502  * If the input is not a string (ex.: `"Hello world!"`) but a char array,
503  * the input message (array) needs to end with NULL (`"\0"`)!
504  *
505  * @param[in] message
506  * The string to print to USARTx.
507  *****************************************************************************/
508 void USART1_print(char *message)
509 {
510  /* "message[i] != 0" makes "uint32_t length = strlen(message)"
511  * not necessary (given string MUST be terminated by NULL for this to work) */
512  for (uint32_t i = 0; message[i] != 0; i++)
513  {
514  /* Wait for TXE flag to be raised */
515  while (!LL_USART_IsActiveFlag_TXE(USART1));
516 
517  /* Write character in Transmit Data register.
518  TXE flag is cleared by writing data in DR register */
519  LL_USART_TransmitData8(USART1, message[i]);
520  }
521 }
