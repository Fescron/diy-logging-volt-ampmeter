199 //
200 // Determine how much we must pad the control block to make it a multiple of a cache line in size
201 // Assuming: U8 = 1B
202 // U16 = 2B
203 // U32 = 4B
204 // U8/U16/U32* = 4B
205 //
206 #if SEGGER_RTT_CPU_CACHE_LINE_SIZE // Avoid division by zero in case we do not have any cache
207  #define SEGGER_RTT__ROUND_UP_2_CACHE_LINE_SIZE(NumBytes) (((NumBytes + SEGGER_RTT_CPU_CACHE_LINE_SIZE - 1) / SEGGER_RTT_CPU_CACHE_LINE_SIZE) * SEGGER_RTT_CPU_CACHE_LINE_SIZE)
208 #else
209  #define SEGGER_RTT__ROUND_UP_2_CACHE_LINE_SIZE(NumBytes) (NumBytes)
210 #endif
211 #define SEGGER_RTT__CB_SIZE (16 + 4 + 4 + (SEGGER_RTT_MAX_NUM_UP_BUFFERS * 24) + (SEGGER_RTT_MAX_NUM_DOWN_BUFFERS * 24))
212 #define SEGGER_RTT__CB_PADDING (SEGGER_RTT__ROUND_UP_2_CACHE_LINE_SIZE(SEGGER_RTT__CB_SIZE) - SEGGER_RTT__CB_SIZE)

221 //
222 // Description for a circular buffer (also called "ring buffer")
223 // which is used as up-buffer (T->H)
224 //
225 typedef struct {
226  const char* sName; // Optional name. Standard names so far are: "Terminal", "SysView", "J-Scope_t4i4"
227  char* pBuffer; // Pointer to start of buffer
228  unsigned SizeOfBuffer; // Buffer size in bytes. Note that one byte is lost, as this implementation does not fill up the buffer in order to avoid the problem of being unable to distinguish between full and empty.
229  unsigned WrOff; // Position of next item to be written by either target.
230  volatile unsigned RdOff; // Position of next item to be read by host. Must be volatile since it may be modified by host.
231  unsigned Flags; // Contains configuration flags
232 } SEGGER_RTT_BUFFER_UP;
233 
234 //
235 // Description for a circular buffer (also called "ring buffer")
236 // which is used as down-buffer (H->T)
237 //
238 typedef struct {
239  const char* sName; // Optional name. Standard names so far are: "Terminal", "SysView", "J-Scope_t4i4"
240  char* pBuffer; // Pointer to start of buffer
241  unsigned SizeOfBuffer; // Buffer size in bytes. Note that one byte is lost, as this implementation does not fill up the buffer in order to avoid the problem of being unable to distinguish between full and empty.
242  volatile unsigned WrOff; // Position of next item to be written by host. Must be volatile since it may be modified by host.
243  unsigned RdOff; // Position of next item to be read by target (down-buffer).
244  unsigned Flags; // Contains configuration flags
245 } SEGGER_RTT_BUFFER_DOWN;

252 typedef struct {

257  SEGGER_RTT_BUFFER_DOWN aDown[SEGGER_RTT_MAX_NUM_DOWN_BUFFERS]; // Down buffers, transferring information down from host via debug probe to target
